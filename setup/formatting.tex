% Basic document formatting setup
\usepackage{setspace}
\setstretch{1} % 1 line spacing

% Ensure Times New Roman and normal weight for body text
\renewcommand{\rmdefault}{ptm}
\renewcommand{\normalsize}{\fontsize{12}{18}\selectfont}

% Chapter formatting - Font size 15 bold, uppercase, centered
\titleformat{\chapter}[block]
{\centering\fontsize{15}{22.5}\bfseries}
{\MakeUppercase{\chaptertitlename\ \thechapter}}{1em}{\MakeUppercase}
\titlespacing*{\chapter}{0pt}{0pt}{12pt}

% Section formatting - Font size 14 bold
\titleformat{\section}
{\fontsize{14}{21}\bfseries}{\thesection}{1em}{}
\titlespacing*{\section}{0pt}{6pt}{6pt}

% Subsection formatting - Font size 13 bold
\titleformat{\subsection}
{\fontsize{13}{19.5}\bfseries}{\thesubsection}{1em}{}
\titlespacing*{\subsection}{0pt}{6pt}{6pt}

% Subsubsection formatting - Font size 13 bold
\titleformat{\subsubsection}
{\fontsize{13}{19.5}\bfseries}{\thesubsubsection}{1em}{}
\titlespacing*{\subsubsection}{0pt}{6pt}{6pt}

% Page setup
\pagestyle{fancy}
\fancyhf{}
\fancyhead[R]{\fontsize{12}{18}\selectfont\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% Remove page numbers from chapter start pages
\fancypagestyle{plain}{
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
}

% Continuation page style for CONTENTS
\fancypagestyle{contentscont}{
\fancyhf{}
\fancyhead[C]{\fontsize{12}{18}\bfseries CONTENTS (Cont'd)}
\fancyhead[R]{\fontsize{12}{18}\selectfont\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
}

% Figure and table numbering
\renewcommand{\thefigure}{\thechapter.\arabic{figure}}
\renewcommand{\thetable}{\thechapter.\arabic{table}}

% Reset figure and table counters at each chapter
\makeatletter
\@addtoreset{figure}{chapter}
\@addtoreset{table}{chapter}
\makeatother

% Caption formatting
\captionsetup{
    format=hang,
    font=small,
    labelfont=bf,
    textfont=normal,
    justification=justified,
    singlelinecheck=false
}

% Paragraph formatting
\setlength{\parindent}{0pt}
\setlength{\parskip}{6pt}

% List formatting
\setlist[itemize]{leftmargin=*}
\setlist[enumerate]{leftmargin=*}

% Table formatting
\renewcommand{\arraystretch}{1.2}
